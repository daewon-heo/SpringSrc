// oracle Spring 계정에 프로시져 만들기

CREATE OR REPLACE PROCEDURE ID_DUP_CHK
( V_ID IN MEMBER.USERID%TYPE,
  PR_RESULT OUT NUMBER)
  IS
  BEGIN
     SELECT COUNT(*) INTO PR_RESULT
     FROM MEMBER
     WHERE USERID = V_ID;
  END;
  /

SET SERVEROUTPUT ON;

DECLARE
     RESULT NUMBER;
BEGIN
     ID_DUP_CHK('admin', RESULT);
     DBMS_OUTPUT.PUT_LINE('RESULT :: ' || RESULT);
END;
/
